239 void ndScene::SubmitJobs(void* const context)
240 {
241  T extJob[D_MAX_THREADS_COUNT];
242  dThreadPoolJob* extJobPtr[D_MAX_THREADS_COUNT];
243 
244  const dInt32 threadCount = GetThreadCount();
245  for (dInt32 i = 0; i < threadCount; i++)
246  {
247  extJob[i].m_owner = this;
248  extJob[i].m_context = context;
249  extJob[i].m_timestep = m_timestep;
250  extJobPtr[i] = &extJob[i];
251  }
252  ExecuteJobs(extJobPtr);
253 }

265 D_INLINE dFloat32 ndScene::CalculateSurfaceArea(const ndSceneNode* const node0, const ndSceneNode* const node1, dVector& minBox, dVector& maxBox) const
266 {
267  minBox = node0->m_minBox.GetMin(node1->m_minBox);
268  maxBox = node0->m_maxBox.GetMax(node1->m_maxBox);
269  dVector side0(maxBox - minBox);
270  return side0.DotProduct(side0.ShiftTripleRight()).GetScalar();
271 }
