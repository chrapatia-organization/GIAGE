122  void Push(dFloat32 val)
123  {
124  for (dInt32 i = 1; i < sizeof(m_initialGuess) / sizeof(m_initialGuess[0]); i++)
125  {
126  m_initialGuess[i - 1] = m_initialGuess[i];
127  }
128  m_initialGuess[sizeof(m_initialGuess) / sizeof(m_initialGuess[0]) - 1] = val;
129  }

131  dFloat32 GetInitiailGuess() const
132  {
133  dFloat32 value = dFloat32(dFloat32(0.0f));
134  dFloat32 smallest = dFloat32(1.0e15f);
135  for (dInt32 i = 0; i < sizeof(m_initialGuess) / sizeof(m_initialGuess[0]); i++)
136  {
137  dFloat32 mag = dAbs(m_initialGuess[i]);
138  if (mag < smallest)
139  {
140  smallest = mag;
141  value = m_initialGuess[i];
142  }
143  }
144  return value;
145  }
